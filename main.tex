\begin{minted}{python}
import numba

@numba.jit(nopython=True)
def iterate(array):
    out = 0.0
    for subarray in array:               # for loops in a Numba-
        for subsubarray in subarray:     # compiled block are
            for item in subsubarray:     # just as fast as C++
                out += item
    return out

print(iterate(listB))
\end{minted}

@numba.jit(nopython=True)
def slices(array):
    # slicing works equally well inside the compiled environment
    return array[:, ::-1, ::2], array[[0, 0, -1, -1], [0, -1, 0, -1], 1:-1]
